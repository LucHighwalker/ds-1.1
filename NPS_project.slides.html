def get_nps(rating):
    if rating != '#ERROR!':
        rating = int(rating)
        if rating > 8:
            return 'Promoter'
        elif rating > 6:
            return 'Passive'
        else:
            return 'Detractor'
    else:
        return np.nan

def clean_track(track):
    if track == 'Apps, Explorer':
        return 'Apps'
    elif track == 'Games, Explorer':
        return 'Games'
    else:
        return track
